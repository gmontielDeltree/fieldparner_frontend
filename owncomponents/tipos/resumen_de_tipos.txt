/**
 *  Recordar que la DB es CouchDB (NoSQL).
 *  Lo siguiente son las definiciones de los tipos de datos principales en TypeScript.
 *  Cada tipo puede pensarse como correspondiente a una "tabla" de una base de datos tradicional SQL.
 *  Los campos que tienen un signo de interrogación (ej textura?:...) son opcionales.
 *  Los corchetes denotan una lista (array) del tipo correspondiente (ej Lote[] es una lista de lotes).
 **/

export interface Campo {
  _id: string;
  _rev?: string;
  uuid: string;
  nombre: string;
  campo_geojson: CampoGeoJson;
  lotes: Lote[];
}

/* Es un GeoJSON */
export interface CampoGeoJson {
  id: string;
  type: "Feature";
  properties: {
    hectareas: number;
  };
  geometry: {
    coordinates: [number, number][][];
    type: "Polygon";
  };
}

export interface Lote {
  id: string;
  type: "Feature";
  properties: {
    nombre: string;
    campo_parent_id: string;
    uuid: string; // Same as id
    hectareas: number;
  };
  geometry: {
    coordinates: [number, number][][];
    type: "Polygon";
  };
}

interface Contratista {
  _id: string;
  _rev?: string;
  labores: Labor[];
  nombre: string;
  uuid: string;
  cuit: string;
  datos_generales: { email: string; direccion: string; telefono: string };
}

interface Labor {
  labor: string;
  uuid: string;
}

export interface Proveedor {
  _id: string;
  _rev?: string;
  uuid: string;
  nombre: string;
  cuit: string;
  direccion: string;
  posicion?: LngLatLike;
  telefono: string;
  obs: string;
  last_updated: LastUpdateTag;
  created: CreatedTag;
}

/* Depositos */
interface Deposito {
  _id: string;
  _rev?: string;
  uuid: string;
  nombre: string;
  contratista_asociado?: Contratista;
  proveedor_asociado?: Proveedor;
  posicion?: LngLatLike; //Lng Lat
  direccion?: string;
  tipo: "virtual" | "fisico";
  car?: string;
  /** iso dates zulu */
  last_updated: LastUpdateTag;
  created: CreatedTag;
  pais?: string;
  archivado: boolean;
}

/* Insumos */
interface Insumo {
  _id: string;
  _rev?: string;
  uuid: string;
  marca_comercial: string;
  principio_activo: string;
  tipo: string;
  subtipo: string;
  unidad: string;
  precio: number;
  se_aplica_a: CultivoAplicacion[];
  cultivo?: Cultivo; // Opcional si es semilla
}

interface Cultivo {
  uuid: string;
  nombre: string;
  langkey: string;
}

interface CultivoAplicacion {
  cultivo: Cultivo;
  uuid: string;
  estadio_desde: string;
  estadio_hasta: string;
  dosis_min: number;
  dosis_max: number;
  dosis_sugerida: number;
}

/* Actividades -> Planificaciones y Ejecuciones */

interface Planificacion {
  _id: string;
  _rev?: string;
  uuid: string;
  ts_generacion: number;
  tipo: string;
  lote_uuid: string;
  contratista: Contratista;

  comentario: string;

  detalles: Detalles;

  color?: string;
  texto?: string;

  attachments?: Attachment[];
  motivos_nota?: any;
}

interface Ejecucion {

  _rev?: string;
  uuid: string;

  ingeniero: Ingeniero;

  estado: string;

  condiciones?: CondicionesEjecucion;

type LineaDosis = {
  uuid: string;
  insumo: Insumo;
  motivos: string[];
  dosis: number;
  total: number;
  precio_estimado: number;
};

type LineaLabor = {
  uuid: string;
  labor: Labor;
  costo: number;
  observacion: string;
};

type LineaDosisEjecucion = {
  uuid: string;
  insumo: Insumo;
  motivos: string[];
  dosis: number;
  total: number;
  precio_estimado: number;
  precio_real: number;
  deposito_origen: Deposito;
};

/* Condiciones planificadas */
interface Condiciones {
  temperatura_min: number;
  temperatura_max: number;
  humedad_min: number;
  humedad_max: number;
  velocidad_min: number;
  velocidad_max: number;
}

interface CondicionesEjecucion {
  temperatura: {
    device: DeviceDetalles;
    value: number;
    planificado: { min: number; max: number };
    distancia: number;
  };
  humedad: {
    device: DeviceDetalles;
    value: number;
    planificado: { min: number; max: number };
    distancia: number;
  };
  velocidad: {
    device: DeviceDetalles;
    value: number;
    planificado: { min: number; max: number };
    distancia: number;
  };
  humedad_suelo: {
    device: DeviceDetalles;
    value: number;
    planificado: { min: number; max: number };
    distancia: number;
  };
  temperatura_min: number;
  temperatura_promedio: number;
  humedad_promedio: number;
  velocidad_promedio: number;
  temperatura_max: number;
  humedad_min: number;
  humedad_max: number;
  velocidad_min: number;
  velocidad_max: number;
}
